% ============ SPEAKER NOTES ============
% Speaker notes are ON by default. Every content frame below ends with a
% \note{...} block; beamer_to_pptx.py reads these and attaches them to the
% corresponding slide's Notes pane in the generated .pptx. Keep the \note
% on the line immediately after \end{frame} so the parser picks it up.
% ============ END SPEAKER NOTES ============

\begin{document}

%----------------------------------------------------------------------
\section{Title}
%----------------------------------------------------------------------
\frame{\titlepage}
\note{Welcome the audience. Introduce yourself and co-authors. State the venue and date briefly.}

%----------------------------------------------------------------------
\section{Outline}
%----------------------------------------------------------------------
\begin{frame}{Outline}
  \begin{enumerate}
    \item Introduction
    \item Methods (with image beside text)
    \item Results
    \item Conclusion and thank you
  \end{enumerate}
\end{frame}
\note{Preview the talk. Mention how long each section will take so the audience knows what to expect.}

%----------------------------------------------------------------------
\section{Introduction}
%----------------------------------------------------------------------
\begin{frame}{Introduction}
  \begin{itemize}
    \item \textbf{Topic}: Short description of the problem.
    \item \textbf{Goal}: What you propose or show.
    \item \textbf{Contribution}: Key takeaway (no bold mid-sentence; use labels only).
  \end{itemize}
\end{frame}
\note{Set context for the problem. Expand each bullet with one or two extra sentences of motivation.}

%----------------------------------------------------------------------
\section{Method Example}
%----------------------------------------------------------------------
% Example: image beside text (create_prez layout)
\begin{frame}{Method Example}
  \vspace{-0.4em}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{itemize}
        \item Main idea in words.
        \item Equation if needed:
        \[
          x^* = \arg\min_x f(x)
        \]
        \item Second point; keep text above footer.
      \end{itemize}
    \end{column}
    \begin{column}{0.47\textwidth}
      \centering
      % Replace with: \includegraphics[width=\linewidth,height=0.58\textheight,keepaspectratio]{figure1}
      \rule{0.8\linewidth}{0.58\textheight}\\[0.5em]
      \small\textit{(Agent replaces this with a figure from Extracted\_figures/)}
    \end{column}
  \end{columns}
\end{frame}
\note{Walk through the method step by step. Point at the figure as you explain each stage.}

%----------------------------------------------------------------------
\section{Results}
%----------------------------------------------------------------------
\begin{frame}{Results}
  \begin{itemize}
    \item \textbf{Summary}: Main result in one line.
    \item \textbf{Comparison}: How it compares to baseline.
  \end{itemize}
\end{frame}
\note{Lead with the best-performing result. Briefly call out the key variable that drove the improvement.}

%----------------------------------------------------------------------
\section{Conclusion}
%----------------------------------------------------------------------
\begin{frame}{Conclusion}
  \begin{itemize}
    \item \textbf{Recap}: What you did.
    \item \textbf{Takeaway}: What the audience should remember.
  \end{itemize}
\end{frame}
\note{Restate the takeaway in one short sentence. Thank collaborators and funders if appropriate.}

%----------------------------------------------------------------------

  \vspace{0.5em}
  \par
  Your Institution
  \end{center}
\end{frame}
\note{Invite questions. Have a slide buffer ready for anticipated questions.}

\end{document}
